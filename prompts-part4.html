================================================================
App name:    Tohfa (spelled "Tofa" in some screen files — canonical spelling is Tohfa)
Backend:     Node.js + SQLite
Auth:        JWT access tokens (15-min expiry) + refresh tokens
             Refresh endpoint: POST /api/auth/refresh
             Token storage: sessionStorage (frontend handles this)
Payments:    Razorpay only
Currency:    All amounts stored in PAISE (1 rupee = 100 paise)

  bio                 TEXT,
  location            TEXT,

TASK 02: DB — listings table
TYPE: DB MIGRATION
================================================================
CREATE TABLE IF NOT EXISTS listings (
  id                  INTEGER PRIMARY KEY AUTOINCREMENT,
  seller_id           INTEGER NOT NULL,
  title               TEXT NOT NULL,
  primary_name        TEXT,             -- "Primary Identification Name" — Common Name + Artistic Medium (max 80 chars)
  description         TEXT,
  category            TEXT,             -- 'Botanical Prints' | 'Pressed Flowers' | 'Sculptures' | 'Home Decor' | 'Wearables' | 'Botanical Art' | 'Curated Seeds' | 'Handmade Ceramic'
  primary_medium      TEXT,             -- 'Crochet' | 'Ceramic' | 'Wood' | 'Fiber' | 'Mixed'
  tags                TEXT,             -- JSON array of strings e.g. '["#BOTANICAL","#HANDMADE"]'
  badges              TEXT,             -- JSON array e.g. '["Handmade","Made to order","Limited drop"]'
  price_paise         INTEGER NOT NULL, -- price in paise
  sku                 TEXT,
  stock_count         INTEGER DEFAULT 0,
  processing_time     TEXT,             -- '1-2 business days' | '3-5 business days' | '1-2 weeks'
  gift_wrap_available INTEGER DEFAULT 0,
  gift_wrap_price_paise INTEGER DEFAULT 5000,   -- ₹50 default
  handwritten_note    INTEGER DEFAULT 0,
  status              TEXT DEFAULT 'draft',     -- 'draft' | 'active' | 'paused' | 'deleted'
  cover_photo_url     TEXT,
  view_count          INTEGER DEFAULT 0,
  sale_count          INTEGER DEFAULT 0,
  listing_score       INTEGER DEFAULT 0,        -- 0-100, computed quality score (shown as 92% in Preview screen)
  weight_grams        REAL,
  length_cm           REAL,
  width_cm            REAL,
  height_cm           REAL,
  shipping_profile_id INTEGER,
  published_at        TEXT,
  created_at          TEXT DEFAULT (datetime('now')),
  updated_at          TEXT DEFAULT (datetime('now')),
  FOREIGN KEY (seller_id) REFERENCES seller_profiles(id) ON DELETE CASCADE
);

CREATE INDEX IF NOT EXISTS idx_listings_seller_id ON listings(seller_id);
CREATE INDEX IF NOT EXISTS idx_listings_status ON listings(status);

QUERY PARAMS:
period: "7d" | "30d" | "90d" | "custom" (default: "7d")
from: ISO date string (required when period=custom)
to:   ISO date string (required when period=custom)
BUSINESS LOGIC:
1. Verify seller JWT; get seller_id from token
2. Look up seller_profiles WHERE user_id = req.user.id
3. Compute KPIs from orders for the given period (SUM, COUNT, etc.)
4. Fetch listings WHERE stock_count <= 5 (low stock threshold) LIMIT 5
5. Fetch recent orders for this seller LIMIT 5, most recent first
6. Fetch active announcements from seller_announcements LIMIT 3
7. Fetch zai_mode_state for this seller
8. Return combined dashboard payload
RESPONSE (errors):
403: { "error": true, "message": "Seller profile not found", "code": "NO_SELLER_PROFILE" }

TASK 42: API — POST /api/seller/become
TYPE: API ROUTE
NEEDED BY: tofa_become_a_seller_landing_page ("Start selling" / "Start your shop" buttons)
================================================================
ENDPOINT:
Method + URL: POST /api/seller/become
Auth required: YES (buyer JWT — converts buyer to seller)
Rate limit: 5/min
REQUEST BODY:
{
  "display_name": "My Artisan Studio",
  "handle": "my_artisan_studio",
  "store_currency": "INR"
}
BUSINESS LOGIC:
1. Verify buyer JWT (user must be logged in)
2. Check user does not already have a seller_profile
3. INSERT INTO seller_profiles with user_id, display_name, handle, platform_fee_pct=8
4. UPDATE users SET role = 'seller' (or add seller role — ⚠️ AMBIGUOUS: check users table schema from Part 3)
5. Seed default shipping profiles (Standard Botanical, Express Sage) for this new seller
6. Return new seller profile + redirect token/hint
RESPONSE (success):
{
  "success": true,
  "data": {
    "seller_id": 99,
    "handle": "my_artisan_studio",
    "store_slug": "my-artisan-studio",
    "onboarding_step": 0,
    "redirect_to": "/seller/studio"
  }
}
RESPONSE (errors):
409: { "error": true, "message": "You already have a seller account", "code": "ALREADY_SELLER" }
400: { "error": true, "message": "Handle already taken", "code": "HANDLE_TAKEN" }
